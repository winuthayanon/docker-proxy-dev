# docker-proxy
nginx-web-dev, nginx-gen-dev and nginx-letsencrypt-dev

# How to install
git clone https://github.com/winuthayanon/docker-proxy-dev

cd docker-proxy-dev

# Setup and run by this command.
./start.sh

# Stop
docker-compose down

# Start
docker-compose up -d

# check docker process by this command.
docker ps
